void ans_go(int dir, int x) { //蚂蚁前进，参数：前进的方向 ；这是第几只蚂蚁 
	mapp[ant[x].x][ant[x].y].is_it_ant = false;
	switch (dir) {
		case 1:
			ant[x].y++;
			break;
		case 2:
			ant[x].x++;
			break;
		case 3:
			ant[x].y--;
			break;
		case 4:
			ant[x].x--;
			break;
	}
	mapp[ant[x].x][ant[x].y].is_it_ant = true;
	leave_pheromone(x);
}
