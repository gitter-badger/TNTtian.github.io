【题解】「P1518」 [USACO2.4]两只塔姆沃斯牛 The Tamworth Two

发布于 2020-05-14

很显然，纯模拟！

看到很多大佬用搜索，本蒟蒻表示：晕惹\QAQ

所以来一篇码量大，但是超好理解的题解\QAQ

因为是模拟，没有什么难度，详细看注释

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#define NR 15
using namespace std;
struct P {//定义人物的结构体
	int x, y, direction;//x和y是坐标， direction代表方向
};
P cow, fj;
int a[NR][NR];
//1↑2→3↓4←
//0. 1* 2fj 3cow

void rotate(int x) {//转弯函数
	if (x == 1) fj.direction++;
	if (x == 2) cow.direction++;
	if (fj.direction == 5) fj.direction = 1;
	if (cow.direction == 5) cow.direction = 1;
}

bool check(int x) {//判断前方有没有障碍物
	if (x == 1) {//当这是fj时
		switch (fj.direction) {//不同的朝向，不同的操作
			case 1:
				if (a[fj.x - 1][fj.y] == 1 || fj.x == 1) return true;
				break;
			case 2:
				if (a[fj.x][fj.y + 1] == 1 || fj.y == 10) return true;
				break;
			case 3:
				if (a[fj.x + 1][fj.y] == 1 || fj.x == 10) return true;
				break;
			case 4:
				if (a[fj.x][fj.y - 1] == 1 || fj.y == 1) return true;
				break;
		}
	}
	if (x == 2) {//cow
		switch (cow.direction) {//同fj
			case 1:
				if (a[cow.x - 1][cow.y] == 1 || cow.x == 1) return true;
				break;
			case 2:
				if (a[cow.x][cow.y + 1] == 1 || cow.y == 10) return true;
				break;
			case 3:
				if (a[cow.x + 1][cow.y] == 1 || cow.x == 10) return true;
				break;
			case 4:
				if (a[cow.x][cow.y - 1] == 1 || cow.y == 1) return true;
				break;
		}
	}
	return false;
}

void go(int x) {//前进
	if (x == 1) {//当fj时
		a[fj.x][fj.y] = 0;//将原来的地点变为'.'
		switch (fj.direction) {
			case 1:
				fj.x--;
				break;
			case 2:
				fj.y++;
				break;
			case 3:
				fj.x++;
				break;
			case 4:
				fj.y--;
				break;
		}
		a[fj.x][fj.y] = 2;//将到达的地方变为‘F’
	}
	if (x == 2) {
		a[cow.x][cow.y] = 0;
		switch (cow.direction) {
			case 1:
				cow.x--;
				break;
			case 2:
				cow.y++;
				break;
			case 3:
				cow.x++;
				break;
			case 4:
				cow.y--;
				break;
		}
		a[cow.x][cow.y] = 3;
	}
}

int main() {
	for (int i = 1; i <= 10; i++)
		for (int j = 1; j <= 10; j++) {
			char c;
			scanf(" %c", &c);
			if (c == '.') a[i][j] = 0;
			if (c == '*') a[i][j] = 1;
			if (c == 'F') {
				a[i][j] = 2;
				fj.x = i;
				fj.y = j;
				fj.direction = 1;
			}
			if (c == 'C') {
				a[i][j] = 3;
				cow.x = i;
				cow.y = j;
				cow.direction = 1;
			}
		}
	int cnt = 0;
	while (1) {//循环进行操作
		cnt++;//操作次数，也就是步数
		if (!check(1)) go(1);//如果前方没有障碍物时前进
		else rotate(1);//否则转弯
		if (!check(2)) go(2);//同上
		else rotate(2);
		if (cow.x == fj.x && cow.y == fj.y) break;//当FJ和COW相遇时，退出循环。
		if (cnt == 600000) {//暴力！当步数为600000时就代表再也相遇不了了
			cout << "0" << endl;//直接输出0
			return 0;//结束程序
		}
	}
	cout << cnt << endl;//输出步数
	return 0;
}
